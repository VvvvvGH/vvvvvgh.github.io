                # [up,down,left,right]
                for pixel in [(x, y - 1), (x, y + 1), (x - 1, y), (x + 1, y)]:

                        if img.getpixel(pixel) == (211, 211, 211, 255) or  img.getpixel(pixel) == (219, 211, 211, 255) or img.getpixel(pixel) == (226, 211, 211, 255):
                            count += 1
                    except IndexError:
                        pass
                if count >= 2:
                    img.putpixel((x, y), (211, 211, 211))
    return img
